PagePlot APIs

double[][] getValues(String expression)

Description: Returns the values of expression.
For example:
double[][] v = plot.getValues("ids*vgs");
v[i][j] is the value of “ids*vgs” of point j curve i
Example 
def plots = cmd.plot("ids_vgs_vbs");
double[][] v = plots[0].getValues("ids");
System.out.println(v[0][0]);

int getCurveCount()

Description: Returns the number of curves.
Example 
def plots = cmd.plot("ids_vgs_vbs");
int curveCount = plots[0].getCurveCount;

PlotCurve getCurve(int index)

Description: Returns the curve with given index.
Example 
def plots = cmd.plot("ids_vgs_vbs");
def curve = plots[0].getCurve(0);

void selectRegion(double lower,double upper)

Description: Clears the selected regions.
Example 
def plots = cmd.plot("ids_vgs_vbs");
plots[0].selectRegion(0,0.5);

void clearRegion()

Description: Clears the selected regions.
Example 
def plots = cmd.plot("ids_vgs_vbs");
plots[0].selectRegion(0,0.5);
plots[0].clearRegion();

public void setCurveSelected(int[] curveIndices, boolean selected)

Description: Sets the curves selected or not.
Example 
def plots = cmd.plot("ids_vgs_vbs");

//set curves whose indexs are 0,2,3 selected.
plots[0] setCurveSelected(new int[]{0,2,3},true);
